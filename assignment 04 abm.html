def Nagel_Schreckenbergmodel(road_len, road_density, prob_hit_brakes, max_velocity, time_iterr):
    #let represent empty cells by -1 and cells filled with cars by 0
    
    start = ([0]*int(road_density * road_len) + [-1]*int((1-road_density) * road_len))
    shuffle(start)
    Initial_road = [start]

    for time in range(time_iterr):
        present_event = Initial_road[-1]
        next_event = [-1]*road_len
        for car in range(road_len):
            if present_event[car] > -1:
                current_vel = present_event[car]
                next_velocity_a = min(current_vel+1, max_velocity) # speed_up to max speed, just behind the next car
                count = 1
                while present_event[((car+count)%road_len)] < 0:
                    count+=1 
                next_velocity_slowdown = min(next_velocity_a, count-1)
                if uniform(0,1) < prob_hit_brakes: # with probability p hit the brakes, otherwise sustain velocity
                    next_vel = max(next_velocity_slowdown-1, 0)
                else:
                    next_vel = next_velocity_slowdown
                next_event[(car+next_vel) % road_len] = next_vel  
        Initial_road.append(next_event)
    road_array = np.zeros(shape=(time_iterr,road_len)) #initial road array
    for car in range(road_len):
        for time in range(time_iterr):
            if Initial_road[time][car] > -1:
                road_array[time,car] = 1
            else:
                road_array[time,car] = 0
    #computing the average velocity
    velocity_summ = 0.0
    velocity_count = 0
    for list1 in Initial_road:
        pos_count = len(list(filter(lambda x: (x >= 0), list1))) 
        sum_operations = sum(list(filter(lambda x: (x >= 0), list1))) 
        velocity_summ+=sum_operations
        velocity_count+=pos_count
    if road_density == 0:
        average_velocity = 5 
    elif road_density == 1:
        average_velocity = 0 
    else:
        average_velocity = velocity_summ/velocity_count
    return (road_array, average_velocity)

def Nagel_Schreckenbergmodel(road_len, road_density, prob_hit_brakes, max_velocity, time_iterr):
    
    road_array = np.zeros(shape=(time_iterr, road_len))  # Defining road_array here to store snapshots

    for time in range(time_iterr):
        present_event = Initial_road[-1]
        next_event = [-1]*road_len
        for car in range(road_len):
            if present_event[car] > -1:
                current_vel = present_event[car]
                next_velocity_a = min(current_vel+1, max_velocity) # speed_up to max speed, just behind the next car
                count = 1
                while present_event[((car+count)%road_len)] < 0:
                    count += 1 
                next_velocity_slowdown = min(next_velocity_a, count-1)
                if uniform(0,1) < prob_hit_brakes:  # with probability p hit the brakes, otherwise sustain velocity
                    next_vel = max(next_velocity_slowdown-1, 0)
                else:
                    next_vel = next_velocity_slowdown
                next_event[(car+next_vel) % road_len] = next_vel  
        Initial_road.append(next_event)
        road_array[time, :] = next_event  # Store the state of the road at this time step

    return road_array
