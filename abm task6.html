def init_graph(G):
    for agent in G.nodes():
        G.nodes[agent]['state'] = 0 #the initial state, the installed fraction f(t)=0 
    return G
def bass(g,p,q):
    g = init_graph(g) #p: Rate of spontaneous adoption,q: Rate of imitation.
    adopt, all_adopted= [], [0]
    population = len(g.nodes())
    while sum([g.nodes[node]['state'] for node in g.nodes()]) != population:
        new_adopters=0
        new_imitators=0
        new_innovators=0  
        for i in range(population):
            agent = rn.choice(list(g.nodes())) 
            if g.nodes[agent]['state']==1: 
                pass
            else:
                innovation = rn.random()
                if innovation < p:    #where p is the rate of sponteneous adoption              
                    g.nodes[agent]['state'] = 1 
                    new_adopters += 1; new_innovators += 1
                else: 
                    imitation = rn.random()
                    neighbors = g.neighbors(agent) 
                    neighbors_states = [g.nodes[node]['state'] for node in neighbors]
                    fraction_of_imitators = sum(neighbors_states)/len(neighbors_states) 
                    if imitation < q*fraction_of_imitators: # where q is the rate of imitation 
                        g.nodes[agent]['state'] = 1
                        new_adopters += 1; new_imitators += 1
        culmulative_sum = sum([g.nodes[node]['state'] for node in g.nodes()])
        all_adopted.append(culmulative_sum); adopt.append(new_adopters)
    return all_adopted, adopt

The graph is initialized using the init_graph function. The simulation loop continues until all nodes have adopted the innovation. At each iteration, a random node is chosen, and its adoption state is updated based on the Bass Diffusion Model rules. The simulation tracks the number of new adopters, new imitators, and new innovators at each step, as well as the cumulative number of adopted nodes. The results are returned as lists containing the cumulative number of adopted nodes and the number of new adopters at each step.

#computing average simulated adopters and cummulative at each time step
def averages(n, time, cumulative_results, adopt_results):
    avg_cummulative = []; avg_adopt = []
    for t in range(time):
        sum_cummulative = 0
        sum_adoption = 0
        for j in range(len(cumulative_results)):
            try:
                sum_cummulative += cumulative_results[j][t]
            except:
                sum_cummulative += n   
            try:
                sum_adoption += adopt_results[j][t]
            except:
                pass
        avg_cummulative.append(sum_cummulative/len(cumulative_results)); avg_adopt.append(sum_adoption/len(adopt_results)) 
    return avg_cummulative,avg_adopt

The averages function calculates the average cumulative adoption and average adoption at each time step based on results from multiple simulations. It iterates over time steps, summing cumulative adoption and adoption values from each simulation. The function returns two lists: avg_cummulative and avg_adopt, representing the averages over all simulations at each time step.

def bassOde(F, time, p1, q1): #F: The state variable at a given time,p1: Rate of spontaneous adoption,q1: Rate of imitation.
    dFdt = (1 - F) * (p1 + q1 * F) #rate of change of the fraction of adopters
    return dFdt

def simulations(g, p, q, runs, n, time):
    cum_adopt_mat = [[0 for i in range(len(p)) ] for j in range(len(q))]
    adopt_mat = [[0 for i in range(len(p)) ] for j in range(len(q))] 
    for i in range(len(p)):
        for j in range(len(q)):
            cumulative_results, adopt_results = [], []
            for _ in range(runs): 
                res1, res2 = bass(g, p=p[i], q=q[i])
                cumulative_results.append(res1)
                adopt_results.append(res2)
            cum_adopt_mat[i][j], adopt_mat[i][j] =averages(n, time, cumulative_results, adopt_results)
    return cum_adopt_mat, adopt_mat
